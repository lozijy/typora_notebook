```shell
LCS Solutiond(X,Y,m,n)
输入:序列X，Y,其中X序列长度为m，Y序列长度为n
输出:序列X，Y最长公共子序列的长度
//加入边界条件
for i <- 1 to m do
	L[i,0]<-0
for j <-1 to n do 
	L[0,j]<-0
	
for j <-1 to m do 
	for j <-1 to n do 
	 if X[i]= Y[j]
	 then 	L[i,j]=L[L-1,j-1]+1
	 		B[i,j]<-"↖"
	 else 	L[i,j]<-max(L[i-1,j],L[i,j-1])
	 		if L[i-1,j]=L[i,j-1]
	 		then	B[i,j]<-"W"
	 		if L[i-1,j]<L[i,j-1]
	 		then B[i,j]<-"↑"
	 		if L[i-1,j]>L[i,j-1]
	 		then B[i,j]<-"←"
输出L[i,j]
```
